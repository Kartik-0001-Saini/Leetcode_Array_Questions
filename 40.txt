Question 40.)--> Combination Sum II
Given a collection of candidate numbers (candidates) and a target number (target), find all unique combinations in candidates where the candidate numbers sum to target.

Each number in candidates may only be used once in the combination.

Note: The solution set must not contain duplicate combinations.

solution.)-->
class Solution {
public:
    set<vector<int>> s;
    void getAllCombination(vector<int>& arr, int target, int idx, vector<int>& combine, vector<vector<int>>& res){
        if(target == 0){
            if(s.find(combine) == s.end()){
                res.push_back(combine);
                s.insert(combine);
            }

            return;
        }

        if(target < 0 || idx == arr.size()){
            return;
        }

        // For single time..
        combine.push_back(arr[idx]);
        getAllCombination(arr, target-arr[idx], idx+1, combine, res);
        //For Backtracking...
        combine.pop_back();
        getAllCombination(arr, target, idx + 1, combine, res); 

    }

    vector<vector<int>> combinationSum2(vector<int>& arr, int target) {
        sort(arr.begin(), arr.end());
        vector<int> combine;
        vector<vector<int>> res;

        getAllCombination(arr, target, 0, combine, res);

        return res;
    }
};
