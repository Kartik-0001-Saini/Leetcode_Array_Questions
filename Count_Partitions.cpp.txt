class Solution {
public:
    int countPartitions(vector<int>& nums) {
        int totSum = 0;
        int n = nums.size();
        for(int i=0; i<n; i++){
            totSum += nums[i];
        }

        int leftArr = 0, count = 0;
        for(int i=0; i<n-1; i++){
            leftArr += nums[i];
            int rightArr = totSum - leftArr;

            if((leftArr%2==0) == (rightArr%2==0)){
                count++;
            }
        }

        return count;
    }
};
