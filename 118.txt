Question 118.) --> Pascal's Triangle
Given an integer numRows, return the first numRows of Pascal's triangle.

In Pascal's triangle, each number is the sum of the two numbers..

solution.) -->
class Solution {
public:

    int fac(int num){
        if(num <=1){
            return 1;
        }
        return num * fac(num - 1);
    }

    vector<vector<int>> generate(int n) {
        vector<vector<int>> res;
        // res.push_back({1});

        for(int i=0; i<n; i++){
            vector<int> ans;
            for(int j=0; j<=i; j++){
                int num = fac(i)/ (fac(i-j) * fac(j));
                ans.push_back(num);
            }
            res.push_back(ans);
        }
        return res;
    }
};

this solution is of O(n^3) so it give me TLE(Time Limit Exceed).
