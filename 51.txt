Question 51.) N-Queens
The n-queens puzzle is the problem of placing n queens on an n x n chessboard such that no two queens attack each other.

Given an integer n, return all distinct solutions to the n-queens puzzle. You may return the answer in any order.

Each solution contains a distinct board configuration of the n-queens' placement, where 'Q' and '.' both indicate a queen and an empty space, respectively.

solution.)
class Solution {
public:

    bool isSafe(vector<string>& board, int row, int col, int n){
        // checking for column
        for(int j=0; j<n; j++){
            if(board[row][j] == 'Q'){
                return false;
            }
        }
        // checking for row
        for(int j=0; j<n; j++){
            if(board[j][col] == 'Q'){
                return false;
            }
        }
        // checking for left diagonal
        for(int i=row, j=col; i>=0 && j>=0; i--, j--){
            if(board[i][j] == 'Q'){
                return false;
            }
        }
        // checking for right diagonal
        for(int i=row, j=col; i>=0 && j<n; i--, j++){
            if(board[i][j] == 'Q'){
                return false;
            }
        }

        return true;
    }

    vector<vector<string>> nQueens(vector<string>& board, int n, int row, vector<vector<string>>& ans){
        // base case where we stop
        // we want to fill column first then we go to row
        if(n == row){
            ans.push_back(board);
            return ans;
        }

        for(int j=0; j<n; j++){
            if(isSafe(board, row, j, n)){
                board[row][j] = 'Q';
                nQueens(board, n, row+1, ans);
                // this is where we are using bactracking if we can't find the answer we backtrack and
                // put it as it is as dot..
                board[row][j] = '.';
            }
        }

        return ans;
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<string> board(n, string(n, '.'));
        vector<vector<string>> ans;

        nQueens(board, n, 0, ans);
        return ans;
    }
};

Time Complexity :- O(n!)
Space Complexity :- O(n^2)
